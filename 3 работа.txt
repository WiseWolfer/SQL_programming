SELECT LAST_NAME, JOB_ID, HIRE_DATE FROM employees
WHERE HIRE_DATE BETWEEN TO_DATE('20.02.1998','DD.MM.YYYY') AND TO_DATE('01.05.1998','DD.MM.YYYY')
ORDER BY HIRE_DATE;
